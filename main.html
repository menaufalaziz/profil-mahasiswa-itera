3.1 Boostrap Resampling

# Statistik yang di-boostrap: rata-rata IPK per klaster

stat_ipk_klaster <- function(data, indicies) {
  d <- data[indicies, ]
  tapply(d$ipk, d$klaster, mean)
}

boot_hasil <- boot(
  data = df,
  statistic = stat_ipk_klaster,
  R = 1000
)

# Statistik yang di-boostrap: rata-rata IPK per klaster

stat_ipk_klaster <- function(data, indicies) {
  d <- data[indicies, ]
  tapply(d$ipk, d$klaster, mean)
}

boot_hasil <- boot(
  data = df,
  statistic = stat_ipk_klaster,
  R = 1000
)
